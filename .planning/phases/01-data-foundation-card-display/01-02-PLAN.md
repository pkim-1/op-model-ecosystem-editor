**`renderCard(repo)` function (per D-06, D-08, D-10):**

Creates and returns a card DOM element. Build entirely with DOM methods (createElement, textContent, appendChild) to prevent XSS from repo descriptions/names.

```javascript
function renderCard(repo) {
  const card = document.createElement('article');
  card.className = 'project-card';
  card.addEventListener('click', () => {
    window.open(repo.html_url, '_blank', 'noopener,noreferrer');
  });
  card.setAttribute('role', 'link');
  card.setAttribute('tabindex', '0');
  card.addEventListener('keydown', (e) => {
    if (e.key === 'Enter' || e.key === ' ') {
      e.preventDefault();
      window.open(repo.html_url, '_blank', 'noopener,noreferrer');
    }
  });

  // Name (h3, most prominent per D-08)
  const name = document.createElement('h3');
  name.className = 'card-name';
  name.textContent = repo.name;
  card.appendChild(name);

  // Description (body text per D-08)
  if (repo.description) {
    const desc = document.createElement('p');
    desc.className = 'card-description';
    desc.textContent = repo.description;
    card.appendChild(desc);
  }

  // Metadata container for topics + language
  const meta = document.createElement('div');
  meta.className = 'card-meta';

  // Topics as chips (per D-06)
  if (repo.topics && repo.topics.length > 0) {
    const topicsDiv = document.createElement('div');
    topicsDiv.className = 'card-topics';
    repo.topics.forEach(topic => {
      const chip = document.createElement('span');
      chip.className = 'topic-chip';
      chip.textContent = topic;
      topicsDiv.appendChild(chip);
    });
    meta.appendChild(topicsDiv);
  }

  // Language badge (per D-06)
  if (repo.language) {
    const lang = document.createElement('span');
    lang.className = 'language-badge';
    lang.textContent = repo.language;
    meta.appendChild(lang);
  }

  card.appendChild(meta);
  return card;
}
```

**`renderGrid(repos)` function -- replaces the stub:**

```javascript
function renderGrid(repos) {
  const grid = document.getElementById('project-grid');
  grid.innerHTML = '';
  const fragment = document.createDocumentFragment();
  repos.forEach(repo => {
    fragment.appendChild(renderCard(repo));
  });
  grid.appendChild(fragment);
}
```
